🔄 DATA FLOW EXAMPLE: Create a Job
═══════════════════════════════════════════════════════════════════════════

USER CLICKS "CREATE JOB" BUTTON
        ↓
COMPONENT calls useCreateJob() hook
        ↓
HOOK returns { mutate, isPending, error }
        ↓
USER SUBMITS FORM with job data
        ↓
COMPONENT calls mutation.mutate(jobData)
        ↓
HOOK calls jobService.create(jobData)
        ↓
SERVICE validates with jobCreateSchema
        ↓
SERVICE calls api.post('/jobs', validatedData)
        ↓
AXIOS INSTANCE:
├─ Adds base URL (/api/v1)
├─ Runs request interceptor
├─ Adds headers
└─ Sends HTTP POST
        ↓
BACKEND receives request
        ↓
BACKEND creates job and responds
        ↓
AXIOS INSTANCE:
├─ Runs response interceptor
├─ Handles errors
└─ Returns data
        ↓
SERVICE returns response
        ↓
HOOK gets response
        ↓
HOOK updates React Query cache
        ↓
HOOK invalidates job lists
        ↓
COMPONENT re-renders with new job
        ↓
USER SEES SUCCESS MESSAGE

'use client';
import { useCreateJob } from '@/hooks/useJobs';
import { JobCreateRequest } from '@/models/job/job.model';

export function CreateJobForm() {
  const createMutation = useCreateJob();

  const handleSubmit = async (formData: JobCreateRequest) => {
    try {
      const newJob = await createMutation.mutateAsync(formData);
      console.log('Job created:', newJob.id);
      // Navigate or show success
    } catch (error) {
      console.error('Failed:', error);
    }
  };

  return (
    <form onSubmit={(e) => {
      e.preventDefault();
      const data = Object.fromEntries(new FormData(e.currentTarget));
      handleSubmit(data as JobCreateRequest);
    }}>
      <input name="title" required />
      <input name="company" required />
      <input name="salaryMin" type="number" required />
      <input name="salaryMax" type="number" required />
      <button disabled={createMutation.isPending}>
        {createMutation.isPending ? 'Creating...' : 'Create'}
      </button>
    </form>
  );
}

'use client';

export function JobDetail({ jobId }: { jobId: string }) {
  const { data: job, isLoading, error } = useJob(jobId);

  if (isLoading) return <p>Loading...</p>;
  if (error) return <p>Error loading job</p>;
  if (!job) return <p>Job not found</p>;

  return (
    <div>
      <h1>{job.title}</h1>
      <p>{job.company}</p>
      <p>{job.description}</p>
      <p>Salary: ${job.salaryMin} - ${job.salaryMax}</p>
      <p>Type: {job.employmentType}</p>
    </div>
  );
}

🔧 ADDING A NEW SERVICE (5-MINUTE GUIDE)

STEP 1: Create Model
File: src/models/newfeature/newfeature.model.ts

import { z } from 'zod';

export const newFeatureSchema = z.object({
  id: z.string(),
  name: z.string().min(1),
  description: z.string().optional(),
});

export type NewFeature = z.infer<typeof newFeatureSchema>;


STEP 2: Create Service
File: src/service/newfeature.service.ts

import api from '@/lib/api';
import { NewFeature, newFeatureSchema } from '@/models/newfeature/newfeature.model';

export const newFeatureService = {
  getAll: async () => {
    const response = await api.get<{ data: NewFeature[] }>('/newfeatures');
    return response.data.data;
  },
  create: async (data: NewFeature) => {
    const validated = newFeatureSchema.parse(data);
    const response = await api.post<{ data: NewFeature }>('/newfeatures', validated);
    return response.data.data;
  },
};


STEP 3: Create Hook
File: src/hooks/useNewFeature.ts

'use client';
import { useQuery, useMutation, useQueryClient } from '@tanstack/react-query';
import { newFeatureService } from '@/service/newfeature.service';

export const useNewFeatures = () => {
  return useQuery({
    queryKey: ['newfeatures'],
    queryFn: () => newFeatureService.getAll(),
  });
};

export const useCreateNewFeature = () => {
  const queryClient = useQueryClient();
  return useMutation({
    mutationFn: (data) => newFeatureService.create(data),
    onSuccess: () => {
      queryClient.invalidateQueries({ queryKey: ['newfeatures'] });
    },
  });
};


STEP 4: Use in Component
File: src/app/newfeatures/page.tsx

export default function Page() {
  const { data, isLoading } = useNewFeatures();

  return (
    <div>
      {isLoading ? 'Loading...' : data?.map(item => <div key={item.id}>{item.name}</div>)}
    </div>
  );
}
